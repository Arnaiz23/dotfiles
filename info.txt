Firacode

change zsh -> chsh -s `which zsh`

oh-my-zsh -> sh -c "$(curl -fsSL https://raw.github.com/robbyrussell/oh-my-zsh/master/tools/install.sh)"

powerlevel10k -> git clone --depth=1 https://github.com/romkatv/powerlevel10k.git ${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}/themes/powerlevel10k

.zshrc -> ZSH_THEME="powerlevel10k/powerlevel10k"

p10k configure

plugins -> 
	plugins=(
	  git common-aliases colored-man-pages zsh-autosuggestions zsh-syntax-highlighting
	)

autosuggestions -> git clone https://github.com/zsh-users/zsh-autosuggestions.git $ZSH_CUSTOM/plugins/zsh-autosuggestions

syntax -> git clone https://github.com/zsh-users/zsh-syntax-highlighting.git $ZSH_CUSTOM/plugins/zsh-syntax-highlighting
